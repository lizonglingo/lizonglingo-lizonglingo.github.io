int Partition(int r[],int first,int end){

    int j=end;
    
    while(i<j){
        while(i<j&&r[i]<=r[j])	j--;

        	int temp=r[i];
            r[i]=r[j];
            r[j]=temp;
            i++;
        }

    }
    return i;
