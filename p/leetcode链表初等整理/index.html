			q->next = p1;

bool hasCycle(struct ListNode *head) {

	struct ListNode *p, *q;
	p = head;
	q = head;
	
	if (p == NULL) return false;
	
	while (p!=NULL&&q!=NULL) {
        if(q->next==NULL)return false;
        if(q->next->next==NULL)return false;
		p = p->next;
		q = q->next->next;
		if (p == q) {
			return true;
		}
	}
	return false;

}
