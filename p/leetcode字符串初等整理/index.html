	prefix[0] = 0;
	int len = 0;
	int i = 1;

	while (i < n) {
		if (pattern[i] == pattern[len]) {
			len++;
			prefix[i] = len;
			i++;
		}

			if (len > 0) {
				len = prefix[len - 1];

				prefix[i] = len;

			}
		}
	}
}
