		return good1.ave_value > good2.ave_value;
	}

typedef struct Node { // ÂàÜÊîØÈôêÁïåÊ≥ïÊ†ëÁªìÁÇπ
	int id;								// ÁªìÁÇπÁºñÂè∑
	int layer;							// ÂΩìÂâçÁªìÁÇπÊâÄÂú®Ê†ëÁöÑÂ±ÇÊï∞

	int current_value = 0;				// ÂΩìÂâçÁªìÁÇπÁöÑÊÄª‰ª∑ÂÄº
	double ub = 0;						// ÂΩìÂâçÁªìÁÇπÁöÑÈôêÁïåÂáΩÊï∞ÂÄº

	int cset[MAX] = { 0 };				// ÂΩìÂâçËÉåÂåÖÁöÑËß£ÂêëÈáè

	bool operator<(const Node& nd) const {		// ÂÆö‰πâ‰ºòÂÖàÈòüÂàóÂá∫ÈòüËßÑÂàôÔºåÈôêÁïåÂáΩÊï∞ÂÄºË∂äÂ§ßË∂äÂÖàÂá∫Èòü
		return ub < nd.ub;
	}
}ND;

long long getSystemTime() {
	timeb t;
	ftime(&t);
	return t.time * 1000 + t.millitm;
}


class Bag {		// Âü∫Á±ªÔºåÁî®‰ΩúËõÆÂäõÊ≥ï

		}

		void brute_force(int layer, int current_value, int current_weight) {
			// layer‰ª£Ë°®ÂõûÊ∫ØÂ±ÇÊï∞ÔºåÂÖ∂‰ΩôÂèÇÊï∞ÂàÜÂà´‰ª£Ë°®ÂΩìÂâç‰ª∑ÂÄº‰ª•ÂèäÂΩìÂâçÈáçÈáè

			if (layer == set_number) {		// if layer bigger than number, over
				// Ê≥®ÊÑèËøô‰∏™Âú∞ÊñπÔºå‰∏çÁî®Â§ß‰∫é ÔºÅÔºÅÔºÅ

				if (current_value > maxValue && current_weight <= capacity) {

					maxValue = current_value;

					for (int i = 0; i < set_number; i++) {
						max_set[i] = current_set[i];

					}
				}
			}
			else
			{
				for (int j = 1; j >= 0; j--) {

					current_set[layer] = j;

					current_weight += (goods[layer].weight * current_set[layer]);
					current_value += (goods[layer].value * current_set[layer]);

					// ÁªßÁª≠ËøõË°åÂ≠êÊ†ëÁöÑÈÄíÂΩí
					brute_force(layer + 1, current_value, current_weight);

					// ÈÄíÂΩíÂá∫Êù•ÂêéÂõûÊªöÊï∞ÊçÆ
					current_weight -= (goods[layer].weight * current_set[layer]);
					current_value -= (goods[layer].value * current_set[layer]);

				}
			}

		}



	public:

		// Ê†πÊçÆË¥ßÁâ©Âçï‰Ωç‰ª∑ÂÄºËøõË°åÊéíÂ∫è
		void sort_set() {		// ËÆ°ÁÆóÊùÉÂÄº

			for (int i = 0; i < set_number; i++) {

		}

class Bag_with_backtrack : public Bag {		// ÁªßÊâø Bag ÁöÑ‰øùÊä§‰ª•ÂèäÂÖ¨Áî®ÊàêÂëò
	
	protected:
		int current_set[MAX] = { 0 };		// define current_set to save current set


		int Bound_sv(int layer, int cv) {

			int i = layer;
			int sv = cv;

			for (i; i < set_number; i++) {
				sv += goods[i].value;		// Ââ©‰ΩôÁâ©ÂìÅÈÉΩË£Ö‰∏ãÊó∂ÁöÑÊúÄÂ§ß‰ª∑ÂÄº
			}

			return sv;

		}
		
		double Bound(int layer,int cv,int cw){
			int i = layer;
			
			double sv = double(cv);
			int cr = capacity - cw;

			while (i<set_number && goods[i].weight <= cr)
			{
				cr -= goods[i].weight;
				sv = sv + goods[i].value;
				i++;
			}

			if (i < set_number)
				sv = sv + goods[i].ave_value * cr;

			return sv;
		}
		

		// Using back track method to get max value
		void BackTrack(int layer, int current_value, int current_weight) {
			// layer‰ª£Ë°®ÂõûÊ∫ØÂ±ÇÊï∞ÔºåÂÖ∂‰ΩôÂèÇÊï∞ÂàÜÂà´‰ª£Ë°®ÂΩìÂâç‰ª∑ÂÄº‰ª•ÂèäÂΩìÂâçÈáçÈáè
			//double expect_sumValue = 0;		// ËÆ∞ÂΩïÊúüÊúõ‰ª∑ÂÄºÊÄªÂíå

			if (layer == set_number) {		// if layer bigger than number, over
				// Ê≥®ÊÑèËøô‰∏™Âú∞ÊñπÔºå‰∏çÁî®Â§ß‰∫é ÔºÅÔºÅÔºÅ

				if (current_value > maxValue) {

					maxValue = current_value;

					for (int i = 0; i < set_number; i++) {
						max_set[i] = current_set[i];

					}
				}
			}
			else
			{
				for (int j = 0; j <= 1; j++) {

					current_set[layer] = j;

					/************************Ââ™Êûù**************************/
					

					if (
						current_weight + current_set[layer] * goods[layer].weight <= capacity
						&& Bound_sv(layer,current_value) >= maxValue && Bound(layer,current_value,current_weight) > maxValue
						)
					{
						// Êõ¥Êñ∞ÂΩìÂâçËÉåÂåÖÁâ©ÂìÅÁöÑÊÄªÈáçÈáèÔºåÁõÆÂâçËÉåÂåÖÁâ©ÂìÅÁöÑÊÄª‰ª∑ÂÄº
						current_weight += (goods[layer].weight * current_set[layer]);
						current_value += (goods[layer].value * current_set[layer]);

						// ÁªßÁª≠ËøõË°åÂ≠êÊ†ëÁöÑÈÄíÂΩí
						BackTrack(layer + 1, current_value, current_weight);

						// ÈÄíÂΩíÂá∫Êù•ÂêéÂõûÊªöÊï∞ÊçÆ
						current_weight -= (goods[layer].weight * current_set[layer]);
						current_value -= (goods[layer].value * current_set[layer]);

					}
				}
			}

		}


	public:

		Bag_with_backtrack(int N) :Bag(N) {
			cout << "ÂõûÊ∫ØÊ≥ïÂØπË±°ÂÆû‰æãÂåñ..." << endl;
		}


		void get_maxValue_by_backTrack(int layer, int current_value, int current_weight) {

			// ÊåâÁÖßÂçï‰Ωç‰ª∑ÂÄºËøõË°åÊéíÂ∫è
			t1 = getSystemTime(); // ÂºÄÂßãËÆ°Êó∂
			sort_set();

			// ‰ΩøÁî®ÂõûÊ∫ØÊ≥ïËøõË°åÊ±ÇËß£
			BackTrack(layer, current_value, current_weight);
			t2 = getSystemTime();	// ÁªìÊùüËÆ°Êó∂
			
			// ÊâìÂç∞Êó∂Èó¥
			sum_time = t2 - t1;
			cout << "ÂõûÊ∫ØÊ≥ïÊ±ÇËß£ËßÑÊ®°‰∏∫ " << set_number << " ÁöÑ0/1ËÉåÂåÖÈóÆÈ¢òÊâÄÈúÄË¶ÅÁöÑÊó∂Èó¥‰∏∫Ôºö" << sum_time << "ms." << endl;

			// ÊâìÂç∞ÈõÜÂêà
			print_set();
		}

		
};		// ÂõûÊ∫ØÊ≥ïÔºåÁªßÊâøÂü∫Á±ªBag


class Bag_with_branchAndbound :public Bag {		// ‰ªé Bag Á±ª‰∏≠Ê¥æÁîü

	protected:
		
		int tree_node_id = 1;		// Ëß£Á©∫Èó¥Ê†ëÁªìÁÇπid 1ÂºÄÂßã
		
		int lower_bound = 0;		// ÁõÆÊ†áÂáΩÊï∞ÁöÑ‰∏ãÁïå

		// Ë¥™ÂøÉÊ≥ïÊ±ÇËÉåÂåÖÂÆπÈáèÁöÑ‰∏ãÁïå
		int greed() {

			int temp[MAX] = { 0 };		// ÂÆö‰πâ‰∏¥Êó∂Êï∞ÁªÑÔºåÁî®Êù•Ë°®Á§∫Ë¥ßÁâ©ÂêëÈáèÔºåÂπ∂ÂàùÂßãÂåñ‰∏∫0
			int cap = capacity;			// ÊöÇÂ≠òËÉåÂåÖÂÆπÈáè
			int w = 0;					// ÂΩìÂâçËÉåÂåÖÂÆπÈáè
			int updown = 0;				// ‰∏ãÁïå
			
			// ÂØπËÉåÂåÖÁâ©ÂìÅÊåâÁÖßÂçï‰Ωç‰ª∑ÂÄºËøõË°åÊéíÂ∫è
			sort_set();

			// Ê±ÇËß£
			for (int i = 0; i < set_number; i++) {

				// Â¶ÇÊûúË∂ÖÈáçÔºåË∑≥Âá∫
				if ((w+goods[i].weight) > cap) {
					continue;
				}

				temp[i] = 1;
				w += goods[i].weight;
				updown += (temp[i] * goods[i].value);
			}

			cout << "Ëß£ÁöÑ‰∏ãÁïåÊòØÔºö" << updown << endl;

			return updown;

		}


		// ÈôêÁïåÂáΩÊï∞Ê±Ç‰∏äÁïå
		void limit(ND &n)				        //ËÆ°ÁÆóÂàÜÊûùÁªìÁÇπÁöÑ‰∏äÁïå
		{
			// ‰∏ã‰∏Ä‰∏™Ë¶ÅÈÄâÁöÑÁâ©ÂìÅÂ∞±ÊòØÁ¨¨layerÂ±ÇÁâ©ÂìÅ
			int w = n.current_weight;
			double v = n.current_value;
			int i = n.layer;

			while (i < set_number && w + goods[i].weight <= capacity) {
				w += goods[i].weight;
				v += goods[i].value;
				i++;
			}

			if (i < set_number) {  // Ë£ÖÈÉ®ÂàÜÁâ©ÂìÅ
				n.ub = v + (capacity - w)*goods[i].ave_value;
			}
			else
			{
				n.ub = v;
			}
		}

		// ‰ºòÂÖàÈòüÂàóËøõÈòüÊìç‰Ωú
		void EnQueue(ND n, priority_queue<ND> &q) {
			if (n.layer == set_number) {		// Â∑≤ÁªèÂà∞ËææÂè∂Â≠êÁªìÁÇπ

				if (n.current_value > maxValue) {		// Êõ¥Êñ∞maxValue
					maxValue = n.current_value;
					for (int i = 0; i < set_number; i++) {
						max_set[i] = n.cset[i];
					}
				}
			}
			else {
				q.push(n);			// ÈùûÂè∂Â≠êÁªìÁÇπËøõÈòü
			}
		}

		// ÂàÜÊîØÈôêÁïåÊ≥ï
		void branchAndbound() {

			// È¶ñÂÖà‰ΩøÁî®Ë¥™ÂøÉÊ≥ïÊ±ÇËß£Ê≠§goodsÈõÜÂêàÁöÑÁõÆÊ†áÂÄº‰∏ãÁïå
			lower_bound = greed();

			priority_queue<ND> q;		// ‰ºòÂÖàÈòüÂàó
			ND n0, n1, n2;				// ÂÖàÂÆö‰πâÊ†πÁªìÁÇπÔºåÂ∑¶Â≠©Â≠êÁªìÁÇπÔºåÂè≥Â≠©Â≠êÁªìÁÇπ

			/*ÂàùÂßãÂåñÊ†πËäÇÁÇπ*/
			n0.id = tree_node_id++;		
			n0.layer = 0;
			n0.current_value = 0;
			n0.current_weight = 0;
			for (int i = 0; i < set_number; i++) {
				n0.cset[i] = 0;
			}
			limit(n0);					// Ê±ÇÊ†πÁªìÁÇπ‰∏äÁïå
			q.push(n0);					// Ê†πÁªìÁÇπËøõÈòü

			while (!q.empty())			// Èòü‰∏çÁ©∫Âæ™ÁéØ
			{
				n0 = q.top();				// ÂèñÈòüÂ§¥
				q.pop();				// Âá∫Èòü
				if (n0.current_weight + goods[n0.layer].weight <= capacity && n0.ub >= lower_bound) {	// Ë∂ÖÈáçÂâ™Êûù Â∞è‰∫éÊúÄÂ∞èÊúüÊúõÈáçÈáèÂâ™Êûù

					// ËÆæÁΩÆÂ∑¶Â≠©Â≠êÁªìÁÇπ
					n1.id = tree_node_id++;
					n1.layer = n0.layer + 1;
					n1.current_weight = n0.current_weight + goods[n0.layer].weight;
					n1.current_value = n0.current_value + goods[n0.layer].value;
					/*Â§çÂà∂Ëß£ÂêëÈáè*/
					for (int i = 0; i < set_number; i++) {
						n1.cset[i] = n0.cset[i];
					}
					n1.cset[n1.layer-1] = 1;			// Êõ¥Êñ∞Ëß£ÂêëÈáè
					limit(n1);						// Ê±ÇÈôêÁïåÂáΩÊï∞‰∏äÁïå
					EnQueue(n1, q);					// Â∑¶Â≠©Â≠êËøõÈòü
				}

				// ËÆæÁΩÆÂè≥Â≠©Â≠êÁªìÁÇπ
				n2.id = tree_node_id++;
